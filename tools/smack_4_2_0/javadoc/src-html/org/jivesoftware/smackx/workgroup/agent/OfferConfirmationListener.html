018
019public interface OfferConfirmationListener {
020
021
022    /**
023     * The implementing class instance will be notified via this when the AgentSession has confirmed
024     * the acceptance of the <code>Offer</code>. The instance will then have the ability to create the room and

026     *
027     * @param confirmedOffer the ConfirmedOffer
028     */
029    void offerConfirmed(OfferConfirmation confirmedOffer);
030}
