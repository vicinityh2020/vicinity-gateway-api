017package org.jivesoftware.smackx.mam.provider;
018
019import org.jivesoftware.smack.provider.IQProvider;
020import org.jivesoftware.smack.util.ParserUtils;
021import org.jivesoftware.smackx.mam.element.MamFinIQ;
022import org.jivesoftware.smackx.rsm.packet.RSMSet;
023import org.jivesoftware.smackx.rsm.provider.RSMSetProvider;
024import org.xmlpull.v1.XmlPullParser;
025
026/**
027 * MAM Fin IQ Provider class.
028 * 
029 * @see <a href="http://xmpp.org/extensions/xep-0313.html">XEP-0313: Message
030 *      Archive Management</a>
031 * @author Fernando Ramirez
032 *
033 */
034public class MamFinIQProvider extends IQProvider<MamFinIQ> {
035
036    @Override
037    public MamFinIQ parse(XmlPullParser parser, int initialDepth) throws Exception {
038        String queryId = parser.getAttributeValue("", "queryid");
039        boolean complete = ParserUtils.getBooleanAttribute(parser, "complete", false);
040        boolean stable = ParserUtils.getBooleanAttribute(parser, "stable", true);
041        RSMSet rsmSet = null;
042
043        outerloop: while (true) {
044            int eventType = parser.next();
045            switch (eventType) {
046            case XmlPullParser.START_TAG:
047                if (parser.getName().equals(RSMSet.ELEMENT)) {
048                    rsmSet = RSMSetProvider.INSTANCE.parse(parser);
049                }
050                break;
051            case XmlPullParser.END_TAG:
052                if (parser.getDepth() == initialDepth) {
053                    break outerloop;
054                }
055                break;
056            }
057        }
058
059        return new MamFinIQ(queryId, rsmSet, complete, stable);
060    }
061
062}
