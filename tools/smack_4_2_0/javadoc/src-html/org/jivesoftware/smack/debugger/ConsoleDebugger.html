017package org.jivesoftware.smack.debugger;
018
019import org.jivesoftware.smack.XMPPConnection;
020
021import java.io.PrintWriter;
022import java.io.Reader;
023import java.io.StringWriter;
024import java.io.Writer;
025import java.text.SimpleDateFormat;
026import java.util.Date;
027
028/**
029 * Very simple debugger that prints to the console (stdout) the sent and received stanzas. Use
030 * this debugger with caution since printing to the console is an expensive operation that may
031 * even block the thread since only one thread may print at a time.<p>
032 * <p/>
033 * It is possible to not only print the raw sent and received stanzas but also the interpreted
034 * packets by Smack. By default interpreted packets won't be printed. To enable this feature
035 * just change the <tt>printInterpreted</tt> static variable to <tt>true</tt>.
036 *
037 * @author Gaston Dombiak
038 */
039public class ConsoleDebugger extends AbstractDebugger {
040    private final SimpleDateFormat dateFormatter = new SimpleDateFormat("HH:mm:ss");
041
042    public ConsoleDebugger(XMPPConnection connection, Writer writer, Reader reader) {
043        super(connection, writer, reader);
044    }
045
046    @Override
047    protected void log(String logMessage) {
048        String formatedDate;
049        synchronized (dateFormatter) {
050            formatedDate = dateFormatter.format(new Date());
051        }
052        // CHECKSTYLE:OFF
053        System.out.println(formatedDate + ' ' + logMessage);
054        // CHECKSTYLE:ON
055    }
056
057    @Override
058    protected void log(String logMessage, Throwable throwable) {
059        StringWriter sw = new StringWriter();
060        PrintWriter pw = new PrintWriter(sw);
061        // CHECKSTYLE:OFF
062        throwable.printStackTrace(pw);
063        // CHECKSTYLE:ON
064        log(logMessage + sw);
065    }
066
067}
